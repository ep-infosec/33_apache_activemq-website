00293         virtual decaf::lang::Pointer<commands::Command> processTransactionInfo(
00294             commands::TransactionInfo* info ) {
00295 
00296             if( info != decaf::lang::Pointer<commands::Command>() ) {
00297                 switch( info->getType() ) {
00298                     case core::ActiveMQConstants::TRANSACTION_STATE_BEGIN:
00299                         return this->processBeginTransaction( info );
00300                     case core::ActiveMQConstants::TRANSACTION_STATE_COMMITONEPHASE:
00301                         return this->processCommitTransactionOnePhase( info );
00302                     case core::ActiveMQConstants::TRANSACTION_STATE_COMMITTWOPHASE:
00303                         return this->processCommitTransactionTwoPhase( info );
00304                     case core::ActiveMQConstants::TRANSACTION_STATE_END:
00305                         return this->processEndTransaction( info );
00306                     case core::ActiveMQConstants::TRANSACTION_STATE_FORGET:
00307                         return this->processForgetTransaction( info );
00308                     case core::ActiveMQConstants::TRANSACTION_STATE_PREPARE:
00309                         return this->processPrepareTransaction( info );
00310                     case core::ActiveMQConstants::TRANSACTION_STATE_RECOVER:
00311                         return this->processRecoverTransactions( info );
00312                     case core::ActiveMQConstants::TRANSACTION_STATE_ROLLBACK:
00313                         return this->processRollbackTransaction( info );
00314                     default:
00315                         throw exceptions::ActiveMQException(
00316                             __FILE__, __LINE__, "Unknown Transaction Info Type.");
00317                 }
